# Convert the columns to numeric values first
data["Horsepower"] = pd.to_numeric(data["Horsepower"], errors='coerce')
data["0-60 MPH Time (seconds)"] = pd.to_numeric(data["0-60 MPH Time (seconds)"], errors='coerce')
data["Price (in USD)"] = pd.to_numeric(data["Price (in USD)"], errors='coerce')  # Convert price to numeric

# Now filter the data
sports_cars = data[
    (data["Horsepower"] > 350) &
    (data["0-60 MPH Time (seconds)"] < 5.0)
].copy()   # <-- prevents pandas warnings

sports_cars["Value Score"] = (
    sports_cars["Horsepower"] / sports_cars["Price (in USD)"]
)

best_value = sports_cars.sort_values("Value Score", ascending=False).head(1)

best_value
